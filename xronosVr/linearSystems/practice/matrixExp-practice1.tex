        \end{smallmatrix}

        \begin{smallmatrix}
            0 \\
            1
        \end{smallmatrix}

        \end{smallmatrix} \right]$ .\\
        $\left[\begin{smallmatrix} \answer{2e^{-t}} \\ \answer{2e^{-t} + e^t} \end{smallmatrix}\right]$

        $\left[\begin{smallmatrix} \answer{2^n} & \answer{n2^{n-1}} \\ \answer{0} & \answer{2^n} \end{smallmatrix}\right]$
