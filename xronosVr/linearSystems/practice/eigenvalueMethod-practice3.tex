\begin{exercise}
    Consider the system
    \begin{equation*}
        \begin{bmatrix} 
            x' \\ 
            y' 
        \end{bmatrix} 
        = 
        \begin{bmatrix} 
            1& -2 \\ 
            5 & -1 
        \end{bmatrix} 
        \begin{bmatrix} 
            x \\ 
            y 
        \end{bmatrix}.
    \end{equation*} %4.5
    \begin{tasks}
        \task Find the general solution.
        \task Sketch the phase portrait for this system.
        \task Solve the IVP with initial conditions $x(0)=1, y(0)=0$, and determine the maximum $x$-coordinate on this trajectory.
    \end{tasks}
\end{exercise}
