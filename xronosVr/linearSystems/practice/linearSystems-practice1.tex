\begin{exercise}
    Consider the third order differential equation
    \begin{equation*}
        y''' + (y'+1)^2 = e^y + \sin(t+1).
    \end{equation*}
    Convert this to a first order system and simplify as much as possible. Can you write this in the form $\vec{x}' = A\vec{x} + \vec{f}$? Why or why not?
\end{exercise}

    \end{tasks}
