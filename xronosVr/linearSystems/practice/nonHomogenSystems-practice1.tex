\begin{exercise}

    $\displaystyle {\vec{x}}' =
    \begin{bmatrix}
        \frac{1}{t} & -1 \\
        1 & \frac{1}{t}
    \end{bmatrix}
    \vec{x} + \begin{bmatrix} t^2 \\ -t \end{bmatrix} .$
    \begin{tasks}
        \task Check that
            $\displaystyle \vec{x}_c = c_1
            \begin{bmatrix}
                t\, \sin t \\
                - t \, \cos t
            \end{bmatrix}
            + c_2
            \begin{bmatrix}
                t\, \cos t \\
                t \, \sin t
            \end{bmatrix}$
            is the complementary solution.
        \task Use variation of parameters to find a particular solution.
    \end{tasks}
\end{exercise}
