\begin{exercise}%
    Find all of the eigenvalues and eigenvectors of the matrix
    \[ 
        \begin{bmatrix}
            1 & 1 \\ 
            -4 & 6 
        \end{bmatrix}. 
    \]
    
    $\lambda_1 = \answer{2}$, $\vec{v}_1 = \begin{bmatrix} \answer{1} \\ 1 \end{bmatrix}$. \\
    $\lambda_2 = \answer{5}$, $\vec{v}_2 = \begin{bmatrix} \answer{1}\\ \answer{4} \end{bmatrix}$
    \begin{problem}
        Check if the eigenvectors for different eigenvalues are orthogonal. 
        \begin{multipleChoice}
            \choice{They are orthogonal.}
            \choice[correct]{They are not orthogonal.}
        \end{multipleChoice}
        \begin{problem}
            Does this make sense based on the original matrix? \wordChoice{\choice{Yes.}\choice[correct]{No.}}
        \end{problem}
    \end{problem}
\end{exercise}
