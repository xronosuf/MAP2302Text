\begin{exercise}
    Let $F(t) = \frac{1}{2} + \sum_{n=1}^\infty \frac{1}{n^2} \cos (n \pi t)$. Find the steady periodic solution to $x'' + 2 x = F(t)$.  Express your solution as a Fourier series.
\end{exercise}

\end{exercise}
